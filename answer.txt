#include <iostream>
#include <string>
#include <vector>

using namespace std;

class Person {
protected:
    string pname;
    string address;
    int phone_no;

public:
    Person() {
        pname = "";
        address = "";
        phone_no = 0;
    }

    void accept_details() {
        cin >> pname >> address >> phone_no;
    }
};

class Employee : public Person {
public:
    int eno;
    string ename;

    Employee() : eno(0), ename("") {}

    void accept_details() {
        Person::accept_details();
        cin >> eno >> ename;
    }
};

class Manager : public Employee {
public:
    string designation;
    string deptname;
    float basic_salary;

    Manager() : designation("Manager"), deptname(""), basic_salary(0.0) {}

    void accept_details() {
        Employee::accept_details();
        cin >> deptname >> basic_salary;
    }
};

int main() {
    int n;
    cin >> n;

    vector<Manager> managers(n);

    Manager highest_salary_manager;
    float highest_salary = -1;

    for (int i = 0; i < n; i++) {
        managers[i].accept_details();
        if (managers[i].basic_salary > highest_salary) {
            highest_salary = managers[i].basic_salary;
            highest_salary_manager = managers[i];
        }
    }

    cout << "Manager with Highest Salary is : " << highest_salary << endl;
    cout << "Manager Name is : " << highest_salary_manager.ename << endl;

    return 0;
}
